#!/usr/bin/gnuplot
set terminal png size 1920,1080 enhanced font 'Arial, 16'
set output 'speedup.png'

set style line 1 linecolor rgb 'black' linetype 1 linewidth 2
set style line 2 linecolor rgb 'red' linetype 1 linewidth 2
set style line 3 linecolor rgb 'blue' linetype 1 linewidth 2
set style line 4 linecolor rgb 'green' linetype 1 linewidth 2

set border linewidth 2
set key top left
set grid
set mytics 1 
set format y "%.2f"
set xlabel "Threads"
set format x "%3.0f"
set ylabel "Speedup"
set xtics 2
set ytics 0.5
set rmargin 4
set tmargin 2
plot "linear.dat" using 1:2 title "linear speedup" with linespoints ls 1,\
"1000.dat" using 1:2 title "n = m = 1000" with linespoints ls 2,\
"2000.dat" using 1:2 title "n = m = 2000" with linespoints ls 3,\
"3000.dat" using 1:2 title "n = m = 3000" with linespoints ls 4
